### 3.1 时间戳基类 - TimeStampedModel

```python
# 从 django.db 导入 models 模块（包含所有数据库字段类型）
from django.db import models
# 导入 Django 设置，用于获取用户模型
from django.conf import settings
# 导入时区工具，用于处理时间
from django.utils import timezone

# 定义一个抽象基类，用于给其他模型继承
class TimeStampedModel(models.Model):
    # 创建时间字段：自动设置为对象创建时的时间
    created_at = models.DateTimeField(auto_now_add=True)
    # 更新时间字段：每次保存对象时自动更新为当前时间
    updated_at = models.DateTimeField(auto_now=True)

    # 内部 Meta 类用于配置模型
    class Meta:
        abstract = True  # 设为抽象模型，不会在数据库创建表
```

**📖 注释含义解释：**
- `#` - Python 单行注释，解释下面代码的作用
- `auto_now_add=True` - 创建对象时自动设为当前时间，之后不再改变
- `auto_now=True` - 每次保存对象时自动更新为当前时间
- `abstract = True` - 抽象基类，只作为其他类的父类使用

### 3.3 分类模型 - Category

```python
# Category 继承 TimeStampedModel，自动获得 created_at 和 updated_at
class Category(TimeStampedModel):
    # 分类名称：字符类型，最大长度100
    name = models.CharField(max_length=100)
    # 分类颜色：字符类型，最大长度20，默认蓝色 #1890ff
    color = models.CharField(max_length=20, default='#1890ff')
    # 用户外键：每个分类属于一个用户
    user = models.ForeignKey(
        settings.AUTH_USER_MODEL,  # 关联 Django 的用户模型
        on_delete=models.CASCADE,  # 用户删除时，其分类也跟着删除（级联删除）
        related_name='categories'  # 反向关联名：user.categories 可获取用户的所有分类
    )

    class Meta:
        ordering = ['-created_at']  # 默认排序：按创建时间倒序（新的在前）
        # 唯一性约束：同一用户不能有两个同名分类
        constraints = [
            models.UniqueConstraint(
                fields=['user', 'name'],  # 组合字段：用户 + 分类名
                name='unique_user_category_name'  # 约束名称
            )
        ]
    
    # 定义对象的字符串表示，在 Django Admin 中显示
    def __str__(self):
        return self.name  # 返回分类名称
```

**📖 注释含义解释：**
- `ForeignKey` - 外键，建立一对多关系（一个用户可以有多个分类）
- `on_delete=models.CASCADE` - 级联删除，用户删除时相关分类也删除
- `related_name` - 反向访问的属性名
- `Meta` 类 - 模型配置类，不创建字段，只配置元数据
- `ordering` - 默认查询排序方式，`-` 表示倒序
- `UniqueConstraint` - 唯一性约束，确保数据不重复

### 3.5 任务模型 - Task

```python
class Task(TimeStampedModel):
    # 定义优先级选项（元组列表）
    PRIORITY_CHOICES = [
        ('HIGH', 'High'),      # 第一个值存入数据库，第二个值显示
        ('MEDIUM', 'Medium'),  # 中等优先级
        ('LOW', 'Low'),        # 低优先级
    ]

    # 任务标题：必须，最长200字符
    title = models.CharField(max_length=200)
    # 任务描述：文本类型，可以为空（数据库null=True，表单blank=True）
    description = models.TextField(blank=True, null=True)
    # 优先级：字符类型，限制只能从 PRIORITY_CHOICES 中选
    priority = models.CharField(
        max_length=10,
        choices=PRIORITY_CHOICES,  # 限制可选值
        default='MEDIUM',          # 默认中等优先级
        db_index=True              # 给这个字段加索引，查询更快
    )
    # 截止日期：可以为空
    due_date = models.DateTimeField(null=True, blank=True, db_index=True)
    # 是否已完成：布尔类型，默认未完成
    completed = models.BooleanField(default=False, db_index=True)
    # 完成时间：可以为空
    completed_at = models.DateTimeField(null=True, blank=True)
    
    # 关联用户（每个任务属于一个用户）
    user = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name='tasks'
    )
    # 关联分类（可以为空）
    category = models.ForeignKey(
        Category,
        on_delete=models.SET_NULL,  # 删除分类时，任务的分类设为NULL，不删除任务
        null=True,                  # 数据库允许为空
        blank=True,                 # 表单允许为空
        related_name='tasks'
    )
    # 多对多关联标签，使用中间表 TaskTag
    tags = models.ManyToManyField(Tag, through='TaskTag', related_name='tasks')

    class Meta:
        ordering = ['-created_at']

    def __str__(self):
        return self.title

    # 自定义方法：标记任务完成
    def mark_completed(self):
        self.completed = True              # 设为已完成
        self.completed_at = timezone.now()  # 记录完成时间
        self.save()                        # 保存到数据库
```

**📖 注释含义解释：**
- `choices` - 限制字段只能取预设的值
- `default` - 默认值
- `db_index=True` - 为字段建立数据库索引，提高查询速度
- `null=True` - 数据库层面允许字段为空值（NULL）
- `blank=True` - 表单验证层面允许不填（主要影响Django Admin）
- `on_delete=models.SET_NULL` - 外键删除时设为NULL，而不是级联删除
- `ManyToManyField` - 多对多字段，一个任务可以有多个标签，一个标签可以用于多个任务
- `through` - 指定自定义的中间表（而不是Django自动生成）

### 3.6 任务-标签中间表 - TaskTag

```python
# 任务和标签的中间表（多对多关系的实现）
class TaskTag(TimeStampedModel):
    # 关联任务
    task = models.ForeignKey(Task, on_delete=models.CASCADE, related_name='task_tags')
    # 关联标签
    tag = models.ForeignKey(Tag, on_delete=models.CASCADE, related_name='task_tags')

    class Meta:
        # 联合唯一：同一任务不能重复关联同一标签
        unique_together = ['task', 'tag']

    def __str__(self):
        return f'{self.task.title} - {self.tag.name}'  # f-string 格式化字符串
```

```python
from django.contrib.auth.models import User

from .models import Category, Tag, Task

# 用户序列化器
class UserSerializer(serializers.ModelSerializer):
    class Meta:
        model = User                          # 指定对应的模型
        fields = ['id', 'username', 'email', 'password']  # 指定要序列化的字段
        extra_kwargs = {'password': {'write_only': True}}  # 额外配置：密码只写，不返回

# 登录序列化器（不是 ModelSerializer，直接继承 Serializer）
class LoginSerializer(serializers.Serializer):
    username = serializers.CharField()  # 用户名字段

# 分类序列化器
class CategorySerializer(serializers.ModelSerializer):
    class Meta:
        model = Category
        fields = ['id', 'name', 'color', 'created_at']
        read_only_fields = ['id', 'created_at']  # 只读字段，不能通过接口修改

# 标签序列化器
class TagSerializer(serializers.ModelSerializer):
    class Meta:
        model = Tag
        fields = ['id', 'name', 'color', 'created_at']
        read_only_fields = ['id', 'created_at']

# 任务序列化器
class TaskSerializer(serializers.ModelSerializer):
    # 自定义字段：用于接收前端传来的标签ID列表（只写，不返回）
    tag_ids = serializers.ListField(child=serializers.CharField(), write_only=True, required=False)
    
    class Meta:
        model = Task
        fields = [
            'id', 'title', 'description', 'priority', 'due_date',
            'completed', 'completed_at', 'category_id', 'tag_ids',
            'created_at', 'updated_at'
        ]
        read_only_fields = ['id', 'created_at', 'updated_at']
```

**📖 注释含义解释：**
- `ModelSerializer` - 自动根据模型生成序列化器字段，简化开发
- `fields` - 指定要包含哪些字段
- `write_only=True` - 字段只用于反序列化（接收数据），不会在响应中返回
- `read_only_fields` - 字段只用于序列化（返回数据），不会接收修改
- `ListField` - 列表类型字段，`child` 指定列表中元素的类型

### 5.2 注册接口 - `backend/api/views.py`

```python
# 导入 DRF 的装饰器和工具
from rest_framework.decorators import api_view, permission_classes
from rest_framework.permissions import AllowAny, IsAuthenticated
from rest_framework.response import Response
from rest_framework import status
from django.contrib.auth.models import User
# 导入密码加密和验证工具
from django.contrib.auth import make_password, check_password
# 导入 JWT 的刷新令牌类
from rest_framework_simplejwt.tokens import RefreshToken
from .models import Category, Tag, Task
from .serializers import UserSerializer, LoginSerializer, CategorySerializer, TagSerializer, TaskSerializer

# @api_view 装饰器：定义接受 POST 请求的API视图
@api_view(['POST'])
# @permission_classes 装饰器：允许任何人访问（不需要登录）
@permission_classes([AllowAny])
def register(request):
    # 1. 创建序列化器，把请求数据传进去
    serializer = UserSerializer(data=request.data)
    # 2. 验证数据是否合法
    if not serializer.is_valid():
        # 不合法返回400错误和错误信息
        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
    
    # 获取验证后的数据
    data = serializer.validated_data
    
    try:
        # 3. 检查用户名是否已存在
        if User.objects.filter(username=data['username']).exists():
            return Response({'error': 'Username already exists'}, status=status.HTTP_400_BAD_REQUEST)
        # 检查邮箱是否已存在
        if User.objects.filter(email=data['email']).exists():
            return Response({'error': 'Email already exists'}, status=status.HTTP_400_BAD_REQUEST)
        
        # 4. 创建新用户（密码用 make_password 加密）
        user = User.objects.create(
            username=data['username'],
            email=data['email'],
            password=make_password(data['password']),  # 加密密码，不存明文
        )
        
        # 5. 为新用户创建预设分类
        preset_categories = ['工作', '个人', '学习', '其他']
        for cat_name in preset_categories:
            Category.objects.create(name=cat_name, user=user)  # 循环创建
        
        # 6. 生成 JWT Token
        refresh = RefreshToken.for_user(user)  # 创建刷新令牌
        user_data = {
            'id': user.id,
            'username': user.username,
            'email': user.email,
            'createdAt': user.date_joined,
        }
        
        # 7. 返回成功响应
        return Response({
            'user': user_data,
            'access': str(refresh.access_token),  # 访问令牌（短期）
            'refresh': str(refresh),             # 刷新令牌（长期）
        }, status=status.HTTP_201_CREATED)  # 201 表示资源创建成功
    
    except Exception as e:
        # 捕获其他异常，返回500错误
        return Response({'error': str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)
```

**📖 注释含义解释：**
- `@api_view(['POST'])` - 装饰器，把普通函数变成API视图，只接受POST请求
- `@permission_classes([AllowAny])` - 允许未登录用户访问这个接口
- `serializer.is_valid()` - 验证请求数据是否符合要求
- `serializer.validated_data` - 获取验证后的数据
- `User.objects.create()` - 模型管理器的create方法，创建一条记录
- `make_password()` - Django 的密码加密函数
- `status.HTTP_201_CREATED` - HTTP状态码201，表示创建成功
- `try-except` - 异常处理结构，捕获可能的错误

### 5.4 获取任务列表接口

```python
# 这个视图同时处理 GET（获取列表）和 POST（创建）请求
@api_view(['GET', 'POST'])
@permission_classes([IsAuthenticated])  # 需要登录才能访问
def tasks_list_create(request):
    user = request.user  # 获取当前登录用户
    
    # ========== 处理 GET 请求：获取任务列表 ==========
    if request.method == 'GET':
        try:
            # 1. 初始查询：只获取当前用户的任务（重要！数据隔离）
            queryset = Task.objects.filter(user=user)
            
            # 2. 按优先级过滤（如果有优先级参数）
            priority = request.query_params.get('priority')  # 获取查询参数
            if priority:
                queryset = queryset.filter(priority=priority)
            
            # 3. 按分类过滤
            category_id = request.query_params.get('categoryId')
            if category_id:
                queryset = queryset.filter(category_id=category_id)
            
            # 4. 按完成状态过滤
            completed = request.query_params.get('completed')
            if completed is not None:
                # 字符串转布尔值
                queryset = queryset.filter(completed=completed.lower() == 'true')
            
            # 5. 搜索（按标题模糊搜索）
            search = request.query_params.get('search')
            if search:
                queryset = queryset.filter(title__icontains=search)  # icontains 不区分大小写
            
            # 6. 排序：按创建时间倒序
            tasks = queryset.order_by('-created_at')
            
            # 7. 序列化所有任务
            result_tasks = [serialize_task_with_relations(t) for t in tasks]  # 列表推导式
            return Response(result_tasks)
        except Exception as e:
            return Response({'error': str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)
    
    # ========== 处理 POST 请求：创建任务 ==========
    elif request.method == 'POST':
        # 1. 验证数据
        serializer = TaskSerializer(data=request.data)
        if not serializer.is_valid():
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
        
        # 2. 获取验证后的数据
        data = serializer.validated_data
        tag_ids = data.get('tag_ids', [])  # 获取标签ID列表，默认为空列表
        
        try:
            # 3. 获取分类（重要：验证分类确实属于当前用户！）
            category = None
            if data.get('category_id'):
                category = Category.objects.filter(id=data['category_id'], user=user).first()
            
            # 4. 创建任务对象
            task = Task.objects.create(
                title=data['title'],
                description=data.get('description'),
                priority=data.get('priority', 'MEDIUM'),
                due_date=data.get('due_date'),
                user=user,  # 重要：设置任务所属用户
                category=category,
            )
            
            # 5. 关联标签（同样验证标签属于当前用户）
            if tag_ids:
                for tag_id in tag_ids:
                    tag = Tag.objects.filter(id=tag_id, user=user).first()
                    if tag:
                        task.tags.add(tag)  # 添加多对多关联
            
            # 6. 返回创建的任务数据
            result_task = serialize_task_with_relations(task)
            return Response(result_task, status=status.HTTP_201_CREATED)
        except Exception as e:
            return Response({'error': str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)
```

**📖 注释含义解释：**
- `request.method == 'GET'` - 判断请求方法
- `request.user` - 获取当前已认证的用户对象
- `request.query_params` - GET请求的查询参数（URL ?后面的部分）
- `request.data` - POST/PUT请求的数据（请求体中的JSON）
- `filter()` - 查询过滤器
- `title__icontains` - Django ORM 查询语法，双下划线表示字段查找类型
- `[... for ...]` - 列表推导式，快速生成列表
- `task.tags.add(tag)` - 添加多对多关联

```typescript
  headers: {
    'Content-Type': 'application/json',  // 请求头：发送JSON格式数据

// ========== 请求拦截器：自动添加 Token ==========
api.interceptors.request.use((config) => {
  // 从 localStorage 中获取访问令牌
  const token = localStorage.getItem('accessToken');
  if (token) {
    // 如果有 token，添加到请求头 Authorization
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;  // 返回配置
});

// ========== 响应拦截器：自动刷新 Token ==========
api.interceptors.response.use(
  (response) => response,  // 成功响应直接返回
  async (error) => {
    // 如果是401错误（未认证/Token过期）
    if (error.response?.status === 401) {
      const refresh = localStorage.getItem('refreshToken');
      if (refresh) {
        try {
          // 尝试刷新 Token
          const response = await api.post('/auth/token/refresh/', { refresh });
          const { access } = response.data;
          // 保存新的访问令牌
          localStorage.setItem('accessToken', access);
          // 更新原请求的认证头
          error.config.headers.Authorization = `Bearer ${access}`;
          return api(error.config); // 重试原请求
        } catch (e) {
          // 刷新失败：清除令牌，跳转到登录页
          localStorage.removeItem('accessToken');
          localStorage.removeItem('refreshToken');
          window.location.href = '/login';
        }
      }
    }
    return Promise.reject(error);  // 其他错误继续抛出
  }
);
```

```typescript
import type { User, Category, Tag, Task } from '../types';
import { authApi, categoriesApi, tagsApi, tasksApi } from '../services/api';

export const login = createAsyncThunk(
  'auth/login',  // action 类型前缀
  async (data: { username: string; password: string }, { rejectWithValue }) => {
    try {
      const response = await authApi.login(data);
      // 保存 Token 到 localStorage
      localStorage.setItem('accessToken', response.data.access);
      localStorage.setItem('refreshToken', response.data.refresh);
      return response.data.user;  // 返回用户信息作为 fulfilled action 的 payload
    } catch (error: any) {
      return rejectWithValue(error.response?.data || error.message);  // 失败时返回错误
    }
  }
);

  'auth/register',

    try {
      const response = await authApi.register(data);
      localStorage.setItem('accessToken', response.data.access);
      localStorage.setItem('refreshToken', response.data.refresh);
      return response.data.user;
    } catch (error: any) {
      return rejectWithValue(error.response?.data || error.message);
    }
  }
);

  'auth/fetchCurrentUser',
  async (_, { rejectWithValue }) => {
    try {
      const response = await authApi.getCurrentUser();

      return rejectWithValue(error.response?.data || error.message);

        state.user = action.payload;  // 成功，设置用户

        state.loading = false;

      .addCase(login.rejected, (state) => {
        state.loading = false;  // 失败

      // 处理 register 的三种状态
      .addCase(register.pending, (state) => {
        state.loading = true;
      })
      .addCase(register.fulfilled, (state, action: PayloadAction<User>) => {
        state.user = action.payload;
        state.isAuthenticated = true;
        state.loading = false;
      })
      .addCase(register.rejected, (state) => {
        state.loading = false;
      })

        state.user = action.payload;
        state.isAuthenticated = true;
        state.loading = false;
      })
```

```python
  多行
  注释
```

3. **TODO** - 标记待完成的工作
4. **不要** - 重复描述显而易见的代码
